import { ProfileEditForm } from '@/components/profile/ProfileEditForm'
import type { ProfileWithRelations } from '@/lib/actions/profile.types'
import { safeReturnTo } from '@/lib/utils/returnTo'

export const dynamic = 'force-dynamic'
export const metadata = { title: 'Edit profile' }

export default async function ProfileEditPage({
  searchParams,
}: {
  searchParams: Promise<{ returnTo?: string }>

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| `?returnTo=` query param → client-side router.push | Validated server-side by `safeReturnTo()` before passing to the form prop; relative-path-only enforcement prevents open redirects. |
| Profile edit form → `/onboarding?step=1` | Client navigation after save; follows the already-validated path. |
| Server layout `showFinishSetup` prop → client NavLinks | Boolean derived from server-authoritative profile fetch; client cannot forge it. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-9-03 | Spoofing (reapplied) | `?returnTo=` on `/profile/edit` | mitigate | `safeReturnTo()` pure helper (Plan 01) validates relative-path-only. Invalid returnTo → undefined → form toasts in place. Phase 3's existing open-redirect tests are not affected. E2E smoke step 19/20 confirms. |
| T-9-12 | Information Disclosure | `onboarding_completed_at` in AppNav select | accept | Column is read via same "Owners see own profile" RLS policy as `display_name` and `avatar_url`. No additional exposure surface. |
| T-9-13 | Tampering | Client forges `showFinishSetup=true` to render the link when it shouldn't | accept | The prop is computed server-side and embedded in the server-rendered HTML. A client that edits the DOM sees the link but clicking it navigates to `/onboarding`, where the middleware / page itself will redirect a completed user to `/directory` (page.tsx `if (profile?.onboarding_completed_at) redirect('/directory')`). No privilege escalation. |
| T-9-14 | Denial of Service | Repeated `?returnTo` roundtrips after save | accept | After save, the form redirects once; the returnTo URL is a one-shot value on the edit page query string. No loop possible. |
